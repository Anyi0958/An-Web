## 微信小程序 wx.request 二次封装

### 使用示例
```javascript
  // 获取应用实例
  const app = getApp();
  
  app.$getRequest({
    url: 'xxxx',
    data: {
      code: 'xxx',
    },
    dataType:false,
    hasToken:false
  }).then(res => {});
```
### 源码
```javascript
// host地址
const host = "http://192.168.0.90:8889/api";

// 请求超时
const timeout = 20000;
const tokenName = 'accessToken';
const dataType = 'json';

// 请求成功统一处理
const requestSuccess = (res)=>{
  console.log('请求成功：',res)
  let { statusCode,data } = res;
  return new Promise((resolve, reject) => {
    if(statusCode===200){

    }else{
      wx.lin.showMessage({
        duration:3000,
        content:data.message,
        type: 'error'
      })
    }
    resolve(data)
  });
};

// 请求失败统一处理
const requestFail = (res)=>{
  console.log('请求失败：',res)
  return new Promise((resolve, reject) => {
    let msg = '连接服务器异常';
    if(res.errMsg==="request:fail timeout"){
      msg = '请求超时'
    }
    wx.lin.showMessage({
      duration:3000,
      content:msg,
      type: 'error'
    })
    resolve(res)
  });
};

/**
 * wx.request请求
 * @param {*} url             请求地址，必填
 * @param {*} method          HTTP 请求方法 默认为 GET
 * @param {*} timeout         超时时间，单位为毫秒
 * @param {*} header          设置请求的 header，header 中不能设置 Referer。 content-type 默认为 application/json
 * @param {*} data            请求参数
 * @param {*} responseType    响应的数据类型
 * @param {*} dataType        返回的数据格式 
 * 自定义参数
 * @param {*} type            请求数据类型 json / form 
 * @param {*} hasToken        是否携带凭证
 */

const request = async (options)=>{
  // 开始加载
  wx.lin.showLoading({
    type:"flip",
  })

  options.url = host + options.url;
  return await new Promise((resolve, reject) => {
    // 发送请求
    wx.request({
      ...options,
      timeout:timeout,
      success: res => {
        requestSuccess(res).then( resData =>{
          resolve(resData)
        })
      },
      fail: err => {
        requestFail(err).then( errData =>{
          reject(errData)
        })
      },
      complete: e => {
        // 停止加载
        wx.lin.hideLoading()
      }
    })
  })
}

/**
 * 请求
 * @param {*} url       请求地址，必填
 * @param {*} data      请求参数
 * @param {*} type      请求数据类型(form json blob...)，默认 json
 * @param {*} hasToken  是否携带凭证，默认携带
 */

const postRequest = ({ url,data,type=dataType,hasToken })=>{
  // 请求数据类型 json / form 
  let header = {
    'content-type' : type.toLowerCase()==='form'?'application/x-www-form-urlencoded;charset=UTF-8':'application/json;charset=UTF-8'
  }
  // 是否携带凭证
  hasToken&&(header[tokenName] = wx.getStorageSync(tokenName));

  return request({ 'method':'POST',url,data,type,hasToken  })
}

const getRequest = ({ url,data,type=dataType,hasToken })=>{
  // 请求数据类型 json / form 
  let header = {
    'content-type' : type.toLowerCase()==='form'?'application/x-www-form-urlencoded;charset=UTF-8':'application/json;charset=UTF-8'
  }
  // 是否携带凭证
  hasToken&&(header[tokenName] = wx.getStorageSync(tokenName));

  return request({ 'method':'GET',url,data,type,hasToken  })
}

const putRequest = ({ url,data,type=dataType,hasToken })=>{
  // 请求数据类型 json / form 
  let header = {
    'content-type' : type.toLowerCase()==='form'?'application/x-www-form-urlencoded;charset=UTF-8':'application/json;charset=UTF-8'
  }
  // 是否携带凭证
  hasToken&&(header[tokenName] = wx.getStorageSync(tokenName));

  return request({ 'method':'PUT',url,data,type,hasToken  })
}

module.exports ={ 
  postRequest,
  putRequest,
  getRequest
}
```
